import 'dart:async';
import 'dart:math' show Random;

main() async {
  print('Compute π using the Monte Carlo method.');
  await for (var estimate in computePi()) {
    print('π ≅ $estimate');
  }
}

/// 生成一系列精度愈来愈准确的 π 值。
Stream<double> computePi({int batch: 1000000}) async* {
  var total = 0;
  var count = 0;
  while (true) {
    var points = generateRandom().take(batch);
    var inside = points.where((p) => p.isInsideUnitCircle);
    total += batch;
    count += inside.length;
    var ratio = count / total;
    // 圆的面积 A 为： A = π⋅r²， 因此 π = A/r²。
    // 所以，对于一个随机的点(x,y) 的取值： x ∈ <0,1>、
    // y ∈ <0,1>， 该点落于单位面积为 1 的圆中的概率接近于
    //  π / 4。所以 π 的取值
    // 应该为:
    yield ratio * 4;
  }
}

Iterable<Point> generateRandom([int seed]) sync* {
  final random = new Random(seed);
  while (true) {
    yield new Point(random.nextDouble(), random.nextDouble());
  }
}

class Point {
  final double x, y;
  const Point(this.x, this.y);
  bool get isInsideUnitCircle => x * x + y * y <= 1;
}

核心目标

Dart 是一个长期的野心勃勃的项目。这些核心目标驱动着我们的设计和决策。

通过核心库和工具打造一个坚固的基石

编程语言离开的核心库和工具是难成大器的。Dart 近年来已经支持很多大型项目了。

让编写通用功能的代码更加容易

应用编程通常遇到常见的问题和异常。 Dart 在设计的时候就考虑这些常见的异常情况，让通用的问题易于解决、常见的异常易于捕获。这就是为何 Dart 支持 async/await、 generators（发生器）、 string interpolation（字符串插值）、 异常检测以及其他功能。

不要给程序员提供惊喜

程序员写的代码以及如何执行应该具有直接的对应关系。魔术(自动的类型纠正、变量作用域提升hoisting、“helper” globals, …) 功能在大型应用中并不是非常友好。

为真实的应用提供稳定务实的解决方案

Dart 可能看起来比较无聊。我们称之为 有效率 和 稳定。我们和使用 Dart 来开发应用的程序员紧密合作来确保我们的决策都是为了长期的运行下去。
